"use strict"
function splt(str){
    const symbols = str.split(" ");
    const numb = +symbols[0];
    const secondNumb = +symbols[2];
    if (symbols[1] === "+"){
    return numb + secondNumb;
    }
    if (symbols[1] === "-"){
    return numb - secondNumb;
    }
};
const str = "8 - 2";
alert(splt(str));
